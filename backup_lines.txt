const char* weekday_name(int wd) {
    static const char* kDays[] = {"Domingo", "Segunda", "Terça", "Quarta",
                                  "Quinta", "Sexta", "Sábado"};
    if (wd < 0 || wd > 6) return "?";
    return kDays[wd];
}
